解决浏览器刷新上404 问题  

    server {  
        listen       8802;  
        #listen       somename:8080;  
        server_name  172.16.3.8;  

        location / {  
            root   html/admin;  
            index  index.html index.htm;  
            try_files $uri $uri/ @router;  
        }  
        location @router {  
             rewrite ^.*$ /index.html last;   
        }  
    }
