###  jendins启动命令：
docker run \
  --name jenkins_docker \
  -p 8091:8080 \
  -u root \
  -v /home/jenkins/jenkins-data:/var/jenkins_home \
  -v /var/run/docker.sock:/var/run/docker.sock \
  -v /opt/apache-maven-3.5.4:/usr/local/maven3.5 \
  -v /usr/java/jdk1.8.0_202:/usr/local/jdk1.8 \
  -d \
  jenkins/jenkins:lts
